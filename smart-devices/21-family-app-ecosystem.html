第二十一章：家庭APP生态 — 多用户/照护者面板/告警通知

角色权限数据可见性操作能力
老人(被照护者)只读设备自身数据心率/血压/位置/药盒状态SOS按钮/吃药确认/音量调节

社区医护(协作方)只读健康数据心率/血压/睡眠趋势发送健康建议/标注异常

数据库 Schema 设计要点
老人手环产品核心表只有5张，遵循最小必要原则。以下是最关键的 users 表设计：
SQL: 核心表结构 
-- 用户表(子女+老人+医护三种角色)
CREATE TABLE users (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  role VARCHAR(20) CHECK (role IN ('caregiver','elderly','nurse')),
  phone VARCHAR(20) UNIQUE NOT NULL,
  password_hash VARCHAR(255) NOT NULL,
  created_at TIMESTAMP DEFAULT NOW()
);

-- 老人档案表
CREATE TABLE profiles (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES users(id),
  device_id VARCHAR(64) UNIQUE,
  first_name VARCHAR(50),
  date_of_birth DATE,
  medical_conditions TEXT[],  -- 慢性病数组:['高血压','糖尿病']
  emergency_contacts JSONB DEFAULT '[]'::jsonb
);

-- 家庭成员关系表(一人多设备/一设备多 caregiver)
CREATE TABLE family_relations (
  id SERIAL PRIMARY KEY,
  family_id UUID NOT NULL,
  user_id UUID REFERENCES users(id),
  role VARCHAR(20) CHECK (role IN ('primary','collaborator','viewer')),
  invited_at TIMESTAMP DEFAULT NOW(),
  accepted_at TIMESTAMP
);

-- 健康数据时序表(Redis存热数据, PG存冷数据)
CREATE TABLE health_events (
  id BIGSERIAL PRIMARY KEY,
  device_id VARCHAR(64) NOT NULL,
  event_type VARCHAR(30),  -- 'heartbeat','fall','gps','medication'
  payload JSONB NOT NULL,
  recorded_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_device_event ON health_events(device_id, recorded_at DESC);
CREATE INDEX idx_event_type ON health_events(event_type) WHERE event_type = 'fall';

步骤二：编写核心 API 端点

文件代码片段说明
src/routes/auth.ts 
export const login = async (req, res) => {
  const { phone, password } = req.body;
  const user = await pool.query(
    'SELECT * FROM users WHERE phone=$1', [phone]);
  if (!user.rows[0]) return res.status(401).json({error:'用户不存在'});
  const valid = await bcrypt.compare(password, user.rows[0].password_hash);
  if (!valid) return res.status(401).json({error:'密码错误'});
  const token = jwt.sign({userId: user.rows[0].id, role: user.rows[0].role},
    process.env.JWT_SECRET, {expiresIn: '7d'});
  res.json({token, role: user.rows[0].role});
};

// 设备上报健康数据 → 入库 + 触发告警检查
router.post('/telemetry', authenticateToken, async (req, res) => {
  const { deviceId, telemetry } = req.body;
  // 批量插入(高性能)
  const values = telemetry.map(t =>
    [`$1`, `$2`, JSON.stringify(t.payload), `NOW()`]);
  const query = `INSERT INTO health_events(device_id, event_type, payload, recorded_at) VALUES ${values.join(',')} ON CONFLICT DO NOTHING`;
  await pool.query(query, [`deviceId`, ...values.flat()]);
  // 同时写入 Redis 用于实时告警
  await redis.lpush(`device:${deviceId}:live`, JSON.stringify(telemetry));
  await redis.expire(`device:${deviceId}:live`, 86400);
  res.json({ok: true});
});
 设备批量上报，同时写入PG持久化和Redis实时流
src/routes/alerts.ts 
// 告警规则引擎(简化版)
const ALERT_RULES = [
  { type: 'fall', threshold: 1, windowMs: 5000,  notifyAll: true },
  { type: 'high_hr', threshold: 120, windowMs: 300000, notifyAll: false },
  { type: 'missed_med', threshold: 1, windowMs: 7200000, notifyAll: false },
];

// 检查最近一条事件是否触发告警
async function checkAlerts(deviceId, event) {
  const rule = ALERT_RULES.find(r => r.type === event.type);
  if (!rule) return;
  const recent = await redis.lrange(`device:${deviceId}:${event.type}`, 0, -1);
  if (recent.length >= rule.threshold) {
    // 触发告警 → 调用推送服务
    await sendPushNotification(deviceId, event.type);
    // 清理队列
    await redis.del(`device:${deviceId}:${event.type}`);
  }
}

步骤三：配置 Firebase 推送通知

推送服务架构
FCM 推送分为三层：P0 紧急告警 同时走 FCM 高优先级 + 阿里云短信(容灾)、P1 重要通知 仅 FCM、P2 普通信息 存入 APP 内消息队列。
FCM 服务端代码 
// services/fcm.ts (Node.js + Firebase Admin SDK)
import { initializeApp, cert } from 'firebase-admin/app';
import { getMessaging } from 'firebase-admin/messaging';

initializeApp({
  credential: cert({
    projectId: 'elderly-care-prod',
    privateKey: process.env.FCM_PRIVATE_KEY,  // 从密钥JSON读取
    clientEmail: process.env.FCM_CLIENT_EMAIL,
  })
});

const messaging = getMessaging();

export async function sendEmergencyAlert({
  token, userId, alertType, deviceId, message
}: { token: string; userId: string; alertType: string;
     deviceId: string; message: string }) {
  await messaging.send({
    token,
    notification: {
      title: '紧急告警',
      body: message,
    },
    data: { alertType, deviceId, urgency: 'P0' },
    android: {
      priority: 'high',
      notification: {
        sound: 'emergency_alert.mp3',  // 自定义铃声
        channelId: 'emergency_alerts',
      }
    },
    apns: {
      payload: {
        aps: { sound: 'emergency_alert.wav', badge: 1 }
      }
    }
  });
}

export async function sendMedReminder(userId, pillInfo) {
  // P2 级别: 仅 APP 内消息 + FCM 低优先级
  await messaging.sendEachForMulticast({
    tokens: await getUserTokens(userId),
    notification: { title: '吃药提醒', body: pillInfo },
    data: { urgency: 'P2' }
  });
}
